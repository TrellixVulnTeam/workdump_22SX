$\df{NegationOnOrder} \abr x > \mathbb{0} \iff -x < \mathbb{0}$
\begin{enumerate}
  \lit $x > \mathbb{0} \implies \ldots$
  \begin{enumerate}
    \lit $\mathbb{0} = (-x) + x > (-x) + \mathbb{0} = -x$ \pipe $\mathbb{0} > -x$ \pipe $-x < \mathbb{0}$    \ann{\rf{OrderedField}}{}
  \end{enumerate}
  \lit $x > \mathbb{0} \implies -x < \mathbb{0}$
  \lit $-x < \mathbb{0} \implies \ldots$
  \begin{enumerate}
    \lit $\mathbb{0} = x + (-x) < x + \mathbb{0} = x$ \pipe $\mathbb{0} < x$ \pipe $x > \mathbb{0}$    \ann{\rf{OrderedField}}{}
  \end{enumerate}
  \lit $-x < \mathbb{0} \implies x > \mathbb{0}$
  \lit $x > \mathbb{0} \implies -x < \mathbb{0} \land -x < \mathbb{0} \implies x > \mathbb{0}$ \pipe $x > \mathbb{0} \iff -x < \mathbb{0}$    \ann{2, 4}{}
\end{enumerate} \vspace{.75mm} \hrule \vspace{.75mm} \ \\

$\df{PositiveFactorPreservesOrder} \abr (x > \mathbb{0} \land y < z) \implies x * y < x * z$
\begin{enumerate}
  \lit $(x > \mathbb{0} \land y < z) \implies \ldots$
  \begin{enumerate}
    \lit $(-y) + z > (-y) + y = \mathbb{0}$ \pipe $z + (-y) = \mathbb{0}$    \ann{\rf{OrderedField}}{}
    \lit $x * \pr{z + (-y)} > \mathbb{0}$ \pipe $x * z + x * (-y) > \mathbb{0}$    \ann{\rf{OrderedField}}{}
    \lit $x * z = \mathbb{0} + x * z = \pr{x * y + -(x * y)} + x * z = \pr{x * y + x * (-y)} + x * z = \ldots$    \ann{\rf{Field}, \rf{NegationCommutativity}}{}
    \lit $x * y + \pr{x * z + x * (-y)} > x * y + \mathbb{0} = x * y$    \ann{\rf{Field}, 1.2}{}
    \lit $x * z > x * y$    \ann{1.3, 1.4}{}
  \end{enumerate}
  \lit $(x > \mathbb{0} \land y < z) \implies x * z > x * y$
\end{enumerate} \vspace{.75mm} \hrule \vspace{.75mm} \ \\

$\df{NegativeFactorFlipsOrder} \abr (x < \mathbb{0} \land y < z) \implies x * y > x * z$

    \lit $(-x) * y < (-x) * z$ \pipe $\mathbb{0} = x * y + (-x) * y < x * y + (-x) * z$ \pipe $\mathbb{0} < x * y + (-x) * z$    \ann{\rf{PositiveFactorPreservesOrder}}{}
    \lit $\mathbb{0} < (-x) * (-y + z)$ \pipe $\mathbb{0} > x * (-y + z)$ \pipe $\mathbb{0} > -(x * y) + x * z$    \ann{NegationOnOrder}{}
    \lit $x * y > x * z$
  \end{enumerate}
  \lit $(x < \mathbb{0} \land y < z) \implies x * y > x * z$
\end{enumerate} \vspace{.75mm} \hrule \vspace{.75mm} \ \\

$\df{SquareIsPositive} \abr (x \neq \mathbb{0}) \implies x * x > \mathbb{0}$

  \lit $(x < \mathbb{0}) \implies \ldots$

  \end{enumerate}
  \lit $(x < \mathbb{0}) \implies x * x > \mathbb{0}$
  \lit $x \neq \mathbb{0} \implies (x > \mathbb{0} \lor x < \mathbb{0}) \implies x * x > \mathbb{0}$ \pipe $x \neq \mathbb{0} \implies x * x > \mathbb{0}$    \ann{\rf{OrderTrichotomy}, 1, 3}{}
\end{enumerate} \vspace{.75mm} \hrule \vspace{.75mm} \ \\

$\df{ReciprocationOnOrder} \abr (0 < x < y) \implies 0 < 1/y < 1/x$

  \begin{enumerate}
